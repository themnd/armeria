36      private final Map<ClientOption<?>, ClientOptionValue<?>> options = new LinkedHashMap<>();
37      private final ClientDecorationBuilder decoration = ClientDecoration.builder();

40      /**

42       */

45      /**
46       * Creates a new instance with the specified base options.
47       */
48      AbstractClientOptionsBuilder(ClientOptions options) {
49          requireNonNull(options, "options");

53      /**
54       * Adds the specified {@link ClientOptions}.
55       */
56      public AbstractClientOptionsBuilder options(ClientOptions options) {
57          requireNonNull(options, "options");
58          options.asMap().values().forEach(this::option);
59          return this;
60      }

62      /**
63       * Adds the specified {@link ClientOptionValue}s.
64       */
65      public AbstractClientOptionsBuilder options(ClientOptionValue<?>... options) {
66          requireNonNull(options, "options");
67          for (ClientOptionValue<?> o : options) {
68              option(o);
69          }
70          return this;
71      }

73      /**
74       * Adds the specified {@link ClientOptionValue}s.
75       */
76      public AbstractClientOptionsBuilder options(Iterable<ClientOptionValue<?>> options) {
77          requireNonNull(options, "options");
78          for (ClientOptionValue<?> o : options) {
79              option(o);
80          }
81          return this;
82      }

84      /**
85       * Adds the specified {@link ClientOption} and its {@code value}.
86       */
87      public <T> AbstractClientOptionsBuilder option(ClientOption<T> option, T value) {
88          requireNonNull(option, "option");
89          requireNonNull(value, "value");
90          return option(option.newValue(value));
91      }
92  
93      /**
94       * Adds the specified {@link ClientOptionValue}.
95       */
96      public <T> AbstractClientOptionsBuilder option(ClientOptionValue<T> optionValue) {
97          requireNonNull(optionValue, "optionValue");
98          final ClientOption<?> opt = optionValue.option();
99          if (opt == ClientOption.DECORATION) {
100             final ClientDecoration d = (ClientDecoration) optionValue.value();
101             d.decorators().forEach(decoration::add);
102             d.rpcDecorators().forEach(decoration::addRpc);
103         } else if (opt == ClientOption.HTTP_HEADERS) {
104             final HttpHeaders h = (HttpHeaders) optionValue.value();
105             setHttpHeaders(h);
106         } else {
107             options.put(opt, optionValue);
108         }
109         return this;
110     }

232     /**
233      * Sets a {@link Function} that remaps an {@link Endpoint} into an {@link EndpointGroup}.
234      * This {@link ClientOption} is useful when you need to override a single target host into
235      * a group of hosts to enable client-side load-balancing, e.g.
236      * <pre>{@code
237      * MyService.Iface client =
238      *     Clients.newClient("tbinary+http://example.com/api",
239      *                       MyService.Iface.class);
240      *
241      * EndpointGroup myGroup = EndpointGroup.of(Endpoint.of("node-1.example.com")),
242      *                                          Endpoint.of("node-2.example.com")));
243      *
244      * MyService.Iface derivedClient =
245      *     Clients.newDerivedClient(client, options -> {
246      *         return options.toBuilder()
247      *                       .endpointRemapper(endpoint -> {
248      *                           if (endpoint.host().equals("example.com")) {
249      *                               return myGroup;
250      *                           } else {
251      *                               return endpoint;
252      *                           }
253      *                       })
254      *                       .build();
255      *     });
256      *
257      * // This request goes to 'node-1.example.com' or 'node-2.example.com'.
258      * derivedClient.call();
259      * }</pre>
260      *
261      * <p>Note that the remapping does not occur recursively but only once.</p>
262      *
263      * @see ClientOption#ENDPOINT_REMAPPER
264      * @see ClientOptions#endpointRemapper()
265      */
266     public AbstractClientOptionsBuilder endpointRemapper(
267             Function<? super Endpoint, ? extends EndpointGroup> endpointRemapper) {
268         requireNonNull(endpointRemapper, "endpointRemapper");
269         return option(ClientOption.ENDPOINT_REMAPPER, endpointRemapper);
270     }

317     public AbstractClientOptionsBuilder addHttpHeader(CharSequence name, Object value) {

353 
354     final ClientOptions buildOptions() {
355         final Collection<ClientOptionValue<?>> optVals = options.values();
356         final int numOpts = optVals.size();
357         final ClientOptionValue<?>[] optValArray = optVals.toArray(new ClientOptionValue[numOpts + 2]);
358         optValArray[numOpts] = ClientOption.DECORATION.newValue(decoration.build());
359         optValArray[numOpts + 1] = ClientOption.HTTP_HEADERS.newValue(httpHeaders.build());
360 
361         return ClientOptions.of(optValArray);
362     }
